create table TBLPERSON
(
  oid                          NVARCHAR(50) default (NEWID()) not null,
  personcode                   NVARCHAR(50) default (' ') not null,
  personaccount                NVARCHAR(50) default (' ') not null,
  personpassword               NVARCHAR(100) default (' ') not null,
  personname                   NVARCHAR(50) default (' ') not null,
  personsex                    int default (0) not null,
  personstatus                 int default (0) not null,
  persontype                   int default (0) not null,
  personofficephone            NVARCHAR(50) default (' ') not null,
  personmobilephone            NVARCHAR(50) default (' ') not null,
  personemail                  NVARCHAR(50) default (' ') not null,
  persondesc                   NVARCHAR(200) default (' ') not null,
  personmemo                   NVARCHAR(50) default (' ') not null,
  cuser							NVARCHAR(40) default (' ') not null,
  cdate							DATETIME default (GETDATE()) not null,
  muser							NVARCHAR(50) default (' ') not null,
  mdate							DATETIME default (GETDATE()) not null,
  mpwdtime						DATETIME default (GETDATE()) not null,
  personpasswordquestion       NVARCHAR(100) default (' ') not null,
  personpasswordanswer         NVARCHAR(100) default (' ') not NULL
  CONSTRAINT PK_TBLPERSON_OID	PRIMARY KEY(oid)
)
